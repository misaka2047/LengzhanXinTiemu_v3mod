GER_event.1 = { #成立西德
	type = country_event
	orphan = yes
	
	title = BRD.1.t
	desc = BRD.1.d
	flavor = BRD.1.f

	event_image = { #事件图像
		texture = "gfx/event_pictures/BRD.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3
	
	trigger = { 
	}

	option = {
		name = BRD.1.a
		default_option = yes
		add_modifier = {
			name = the_country_of_our_ancestors_modifier
			years = 1
		}
		if ={
            limit ={
                OR ={
                    AND ={
                        exists =C:RHN
                        this=c:RHN
                        c:RHN ?= {is player= yes}
			}
                    AND ={
                        exists =C:HAN
                        this=c:HAN
                        c:HAN ?= {is player= yes}
			}
		}
	}
			play_as = c:BAV
}
	c:BAV = { change_tag = BRD 
		          annex = c:RHN
				  annex = c:HAN
				  make_independent = yes}
	set_capital = STATE_NORTH_RHINE
		create_character = {
			ruler = yes
			}
		create_character = {
			is_general = yes
			ig_leader = yes
			interest_group = ig_armed_forces #军队
		}
	}

	option = {
		name = BRD.1.b
		add_modifier = {
			name = the_country_of_our_ancestors_modifier
			years = 1
		}
		if ={
            limit ={
                OR ={
                    AND ={
                        exists =C:RHN
                        this=c:RHN
                        c:RHN ?= {is player= yes}
			}
                    AND ={
                        exists =C:HAN
                        this=c:HAN
                        c:HAN ?= {is player= yes}
			}
		}
	}
			play_as = c:BAV
}
	c:BAV = { 
			change_tag = BRD 
		        annex = c:RHN
			annex = c:HAN
			}
	set_capital = STATE_NORTH_RHINE
		create_character = {
			ruler = yes
			}
		create_character = {
			is_general = yes
			ig_leader = yes
			interest_group = ig_armed_forces #军队
		}
	c:USA = {
		          create_diplomatic_pact = {
				  country = c:BRD 
				  type = protectorate
			}
		}
	}

	option = {
		name = BRD.1.c
	}
}
