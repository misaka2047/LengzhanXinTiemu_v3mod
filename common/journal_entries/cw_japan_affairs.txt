je_japan_occupied = {
    icon = "gfx/interface/icons/event_icons/waving_flag.dds"
    group = je_group_historical_content
	complete = {
		army_size < 5
        has_law = law_type:law_non_militarized_country
		game_date > 1953.1.1
		NOT = {
			has_journal_entry = je_japan_mac
		}
	}
    on_complete = {
        trigger_event = {
            id = japan_mac.16
        }
    }
	fail = {
		OR = {
			radical_fraction = {
				value >= 0.75
			}
			any_civil_war = {
				civil_war_progress > 0.6
			}
		}
	}
	on_fail = {
		# 之后整点吃鸡大赛剧情
		trigger_event = {
			id = japan_breakdown
		}
	}
	weight = 10000
	should_be_pinned_by_default = yes
}
