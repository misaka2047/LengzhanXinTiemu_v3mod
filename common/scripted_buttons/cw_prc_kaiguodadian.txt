xinanjiaofei_button_key = {

	name = "xinanjiaofei_button_key"
	desc = "xinanjiaofei_button_key_DESC"
	
	
	visible = { always = yes }

	ai_chance = {
		base = 10
	}

	possible = {
		var:xinanfeihuan_start_var = 0
	}
	effect = {
		add_journal_entry = { type = je_xinanjiaofei }
		change_variable = {
			name = xinanfeihuan_start_var
			add = 1
		}
	}
}

kaiguodadadian_button_key = {

	name = "kaiguodadadian_button_key"
	desc = "kaiguodadadian_button_key_DESC"
	
	
	visible = { always = yes }

	ai_chance = {
		base = 10
	}

	possible = {
		var:kaiguodadadian_var = 0

	}

		trigger_event = { id = kaiguodadian.1 days = 0 popup = yes }

		}
	}
