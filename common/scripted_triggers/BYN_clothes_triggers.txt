dahe_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		exists = home_country
     	home_country = {
     		has_law = law_type:law_monarchy
     	}
		OR = {
			AND = {
				culture = {
					has_discrimination_trait = yamato
				}
				OR = {
					NOT = {
						exists = c:riben
					}
					AND = {
						exists = c:riben
						NOT = { home_country = c:riben }
					}
				}
			}
			AND = {
				exists = c:riben
				home_country = {
					this = c:riben
					NOT = { has_variable = meiji_restoration_complete }
				}
			}
		}
	}
}
